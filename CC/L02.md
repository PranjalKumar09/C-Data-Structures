
	Checks if a character:
    
isdigit(c) is a decimal digit.	'0' to '9'
isalpha(c)	Checks if a character is an alphabetic letter.	'a'-'z', 'A'-'Z'
isalnum(c)	Checks if a character is alphanumeric (a digit or a letter).	'a'-'z', 'A'-'Z', '0'-'9'
islower(c)	Checks if a character is a lowercase letter.	'a'-'z'
isupper(c)	Checks if a character is an uppercase letter.	'A'-'Z'
isspace(c)	Checks if a character is a whitespace.	' ', '\t', '\n', '\r', '\v', '\f'
ispunct(c)	Checks if a character is a punctuation character (not alnum, not space).	! " # $ % & ' ( ) * + , - . / : ; < = > ? @ [ \ ] ^ _ { `	}~`
isxdigit(c)	Checks if a character is a hexadecimal digit.	'0'-'9', 'a'-'f', 'A'-'F'

    find prev & next smallest
    vector<int> left(n), right(n); // Store indices of previous and next smaller elements


    for (int i = 0; i < n; ++i) {
        // Length of the window where arr[i] is the minimum
        int len = right[i] - left[i] - 1;
        // Update result for this window size with arr[i]
        if (len > 0 && len <= n) 
            result[len - 1] = max(result[len - 1], arr[i]);
        
    }
    
    for (int i = n - 2; i >= 0; --i) 
        result[i] = max(result[i], result[i + 1]);
